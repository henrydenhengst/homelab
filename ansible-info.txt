# =================================================================
# 🚀 Henry's Enterprise Homelab Playbook
# Doel: Volledige IaC uitrol van OS-hardening tot Home Automation
# =================================================================

- name: Configureer Homelab Server
  hosts: all
  become: yes
  
  roles:
    # -----------------------------------------------------
    # 1. 🏗️ De Fundering (OS & Security)
    # -----------------------------------------------------
    - role: common             # Ballast verwijderen & Sudo setup
    - role: security           # SSH hardening & OS updates
    - role: fail2ban           # Brute-force protectie
    - role: docker             # Docker-CE Engine & Compose installatie
    - role: portainer          # Visueel container beheer

    # -----------------------------------------------------
    # 2. 💾 Opslag, Data & Communicatie
    # -----------------------------------------------------
    - role: postgres           # Database voor je apps
    - role: mosquitto          # MQTT Broker (De ruggengraat voor sensoren)
    - role: rsync              # Lokale synchronisatie/backup
    - role: rclone             # Cloud-offsite backups

    # -----------------------------------------------------
    # 3. 🔍 Hardware & Bridges (Voorwaardelijk)
    # -----------------------------------------------------
    - role: hardware_discovery # Detecteert USB/PCI apparaten
    - role: zigbee2mqtt
      when: zigbee_device is defined
    - role: zwavejsui
      when: zwave_device is defined
    - role: rtl_433
      when: rf_sdr_device is defined
    - role: matter_server
      when: matter_device is defined

    # -----------------------------------------------------
    # 4. 🧠 Het Brein & Dashboards
    # -----------------------------------------------------
    - role: homeassistant      # Central Smart Home Controller
    - role: esphome            # Voor je eigen ESP-sensoren
    - role: flame              # Startpagina / Dashboard

    # -----------------------------------------------------
    # 5. 🛡️ De Poortwachter (Netwerk & SSL)
    # -----------------------------------------------------
    - role: duckdns            # Dynamische DNS update
    - role: caddy              # Reverse Proxy met Auto-SSL

    # -----------------------------------------------------
    # 6. 📊 De Rapporteur (De Finale)
    # -----------------------------------------------------
    - role: audit              # Genereert PDF-audit op laptop ($USER)

    echo "USB schijf gevonden. Start synchronisatie..."
    
    # Backup van je containers en data

  failed_when: false

  set_fact:
    lynis_score: "{{ score_check.stdout | default('66') }}"
